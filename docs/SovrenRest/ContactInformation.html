def initialize(data)
  parse_name(data['PersonName'] || {})
  contact_method = data['ContactMethod'] || []
  parse_email(contact_method)
  parse_websites(contact_method)
  parse_phone_numbers(contact_method)
  parse_addresses(contact_method)
end
